Vocabulary}\begin{itemize}\item[] {\bf \ChapterDDS}\item[$\RR$] The {\em real numbers}.\item[$\NN$] The {\em natural integers}.\item[$\times$] The product of sets or the product of a number by something.\item[$\Def(f)$] The {\em set of definition\/} of the map $f$.\item[$\Val(f)$] The values of the map $f$.\item[$\Maps({\X},{\Y})$] The maps from ${\X}$ to ${\Y}$. \item[$\id_\X$] The identity map of $\X$. \item[$n$-domain] A (non empty) open set of the vector space $\RR^n$.\item[domain] An arbitrary $n$-domain, for some $n$.\item[$\Domains(\RR^n)$] The  domains of $\RR^n$. \item[$\Domains$] The  domains of the $\RR^n$, $n$ running over $\NN$. \item[$\Powerset(\E)$] The {\em powerset} of the set $\E$, that is, the set of all the subsets of the set $\E$.\item[$\dim(\U)$] The {\em dimension\/} of the domain $\U$, that is, if $\U \subset \R^n$, $\dim(\U) = n$.\item[Parametrization] Any map defined from a domain to some set.\item[$\Param(\X)$] The parametrizations of a set ${\X}$. \item[$\Param(\U,\X)$] The parametrizations of a set $\X$ defined on a domain $\U$. \item[$\bmx$] A bold lower case letter, a constant map (here with value $x$).\item[$\D(\F)(s)$ or $\D(\F)_s$ or ${\partial r \over \partial s}$] For any smooth map $\F : s \mapsto r$between real domains, the tangent linear map of ${\F}$ at the point $s$.\item[$\pr_k$] From a product, the projection on the $k$-th factor.\item[$\cD$] Refers to a diffeology.\item[$\D1, D2, D3$] Refers to the three axioms of diffeology.\item[$\cD(\U,\X)$] The plots, defined on $\U$, of the space $\X$.\item[$\cD(\X,\X')$ or $\Cinfty(\X,\X')$] The smooth maps from $\X$ to $\X'$\item[$\Diff(\X,\X')$] The diffeomorphisms from $\X$ to $\X'$.\item[$\Set$, $\Diffeology$ etc.] Names of categories.\item[$\discrete{\cD}(\X)$] The discrete diffeology of ${\X}$.\item[$\X_\circ$] $\X$ equipped with the discrete diffeology.\item[$\coarse{\cD}(\X)$] The coarse diffeology of ${\X}$.\item[$\X_\bullet$] $\X$ equipped with the coarse diffeology.\item[$f^*(\cD')$] The pullback of the diffeology $\cD'$ by $f$.\item[$\coprod_{i \in \cI} \X_i$] The sum (coproduct) of diffeological spaces.\item[$\class(x)$] The equivalence class of $x$.\item[$f_*(\cD)$] The pushforward of the diffeology $\cD$ by $f$.\item[$\prod_{i \in \cI} \X_i$] The product of diffeological spaces.\item[$\Paths(\X)$] The space of paths in $\X$, that is, $\Cinfty(\R,\X)$.\item[$\DPaths{k}(\X)$] The space of iterated $k$-paths in $\X$, that is, $\Cinfty(\R,\DPaths{k-1}(\X))$.\item[$\delta x$] Any vector of $\RR^n$, when $x$ is a generic point in an $n$-domain.\item[$\gen{\cF}$] The diffeology generated by a family $\cF$ of parametrizations.\item[$\Nebula(\cF)$] The nebula of the family of parametrizations $\cF$.\item[$\dim(\X)$] The global dimension of $\X$, for a diffeological space.\item[$\Gen(\X)$] The generating families of the diffeology of $\X$.\item[]\item[] {\bf \ChapterLAD}\item[$\Cinfty_\loc(\X,\X')$] The locally smooth maps from $\X$ to $\X'$.\item[$\Diff_\loc(\X,\X')$] The locally diffeomorphisms from $\X$ to $\X'$.\item[$=_\loc$] Is equal/coincides locally.\item[$\germ$] The germ of a map.\item[D-open] Open for the D-topology.\item[$\dim_x(\X)$] The pointwise dimension of $\X$ at $x$.\item[]\item[] {\bf \ChapterDVS}\item[$\DLin(\E,\E')$] The smooth linear maps from $\E$ to $\E'$.\item[$\cH$, $\cH_\RR$, $\cH_\CC$] The standard Hilbert space, on $\RR$, on $\CC$.\item[$\X \cdot \Y$] The scalar or Hermitian product.\item[]\item[] {\bf \ChapterMSME}\item[$\cS_\RR$, $\cS_\CC$] The infinite Hilbert sphere, in $\cH_\RR$ or $\cH_\CC$.\item[$\cP$, $\cP_\CC$] The infinite projective space.\item[$\HH_n$, $\KK_n$] The half $n$-space or the $n$-corner.\item[]\item[] {\bf \ChapterHODS}\item[$\0(\gamma)$, $\1(\gamma)$, $\ends(\gamma)$] Starting point, ending point and ends of a paths $\gamma$.\item[$\Paths(\X,\A,\B)$] The paths in $\X$, starting in $\A$ and ending in $\B$.\item[$\Loops(\X)$] Loops in $\X$\item[$\Loops(\X,x)$, $\Loops_n(\X,x)$] Loops and iterated loops in $\X$, based at $x$.\item[$\gamma \vee \gamma'$] Concatenation of paths.\item[$\rev(\gamma)$] The reverse path of $\gamma$.\item[$\stPaths$, $\stLoops$] Stationary paths, loops.\item[$\gamma^\star$] Smashed path, making it stationary.\item[$\pi_0(\X)$, $\pi_k(\X,x)$] The components of $\X$ and $k$th-homotopy groups based at $x$.\item[$\pi_k(\X,\A,x)$] Telative homotopy pointed-sets/groups.\item[$\Pi(\X)$] The Poincar\'e groupoid of $\X$.\item[]\item[] {\bf \ChapterDFADRC}\item[${\E}*{\F}$ or $\Lin(\E,\F)$] The space of linear maps from ${\E}$ to ${\F}$.\item[${\E}^*$] Dual vector space ${\E}$, that is, ${\E}*\RR$.\item[Vector --- Covector] An element of a vector space --- An element of its dual.\item[$\otimes$, $\wedge$] Tensor product, exterior product.\item[$\sgn(\sigma)$] Signature of the permutation $\sigma$.\item[$\DLin(\E,\F)$, ${\E}^*_\infty$] Smooth linear maps, smooth dual.\item[$\M^*(\A)$] Pullback of a covariant tensor $\A$ by a linear map $\M$.\item[$\vol$, $\vol_\cB$] Volume, canonical volume of a basis $\cB$.\item[$\Lambda^k({\E})$] The space of all $k$-linear forms on a vector space ${\E}$.\item[$\DForms^k(\X)$] The space of differentiable $k$-forms on a diffeological space $\X$.\item[$\Forms^{k}_x(\X)$] The space of the values of the $k$-forms of $\X$ at the point $x$.\item[$\alpha(\P)$] The value of the differential form $\alpha$ on the plot $\P$.\item[$f^*(\alpha)$] The pullback of the differential form $\alpha$ by the smooth map $f$.\item[$d\alpha$] The {\em exterior derivative} of the differential form $\alpha$.\item[$\theta$] The canonical $1$-form on $\RR$ or on an irrational torus $\RR/\Gamma$.\item[$\delta \P$, $\delta\sigma$, $\delta c$] The {\em variation} of a plot $\P$, a cube $\sigma$, a chain $c$.\item[$\alpha \rfloor \delta \sigma$] The {\em contraction} of a $k$-form $\alpha$ by a variation of a cube $\sigma$,a differential $(k-1)$-form on $\Dom(\sigma)$.\item[$i_{\F}$] The {\em contraction} by an arc of diffeomorphisms $\F$.\item[$\DLie_\F$] The Lie derivative by an arc of diffeomorphisms $\F$.\item[$\DCubes{p}(\X)$, $\DChains{p}(\X)$] The smooth $p$-cubes, $p$-chains, in $\X$.\item[$\ZG_\star(\X,\cdot)$, $\ZG^\star(\X,\cdot)$] The cycles/cocycles groups with coefficients.\item[$\BG_\star(\X,\cdot)$, $\BG^\star(\X,\cdot)$] The boundaries/coboundaries groups with coefficients.\item[$\HG_\star(\X,\cdot)$, $\HG^\star(\X,\cdot)$] The homology/cohomology groups with coefficients.\item[$\displaystyle{\int_\sigma\alpha}$, $\displaystyle\int_c\alpha$] The integral of a differential form on a cube/cubic-chain.\item[Magma] A set $\A$ equipped with an internal operation $\star$.\item[$\CHK$] The Chain-Homotopy operator.\item[$\eL(\gamma)$, $\eR(\gamma)$] Pre or post concatenation by $\gamma$.\item[]\item[] {\bf \ChapterDG}\item[$\DHom(\G,\G')$] The smooth homomorphisms from $\G$ to $\G'$.\item[$\Ab{\G}$] The Abelianized group $\G/[\G,\G]$.\item[$\eR(x)$ or $\hat x$] The orbit map from $\G$ to $\X$.\item[$\eL(g)$, $\eR(g)$] The left/right multiplication by $g$ in $\G$.\item[$\Ad(g)$] The Adjoint action of $g$ on $\G$, $\Ad(g) = \eL(g) \circ \eR(g^{-1})$.\item[{\em Coset}] The left or right orbits of a subgroup in a group.\item[$\cG^*$] The {\em space of momenta} of a diffeological group $\G$.\item[$\Ad_*$, $\Ad_*^{\Gamma,\theta}$] The coadjoint action of $g$ on $\cG^*$, on $\cG^*/\Gamma$ with cocycle $\theta$.\item[]\item[]{\bf \ChapterDFB}\item[$\source$, $\target$, $\car$] The source, target and characteristic maps on a groupoid.\item[$f^*(\T)$] The pullback by $f$ of the total space $\T$ of a projection.\item[$\T \times_\G \E$ or $\T \times_\rho \E$] An associate bundle.\item[$\Theta$] A connection on a principal bundle.\item[$\HorPaths(\Y,\Theta)$] The horizontal paths in $\Y$, for the connection $\Theta$.\item[$\T_{\Gamma}$] A {\em torus} $\RR^n/\Gamma$, $\Gamma$ discrete (diffeologically) and generator of $\RR^n$.\item[$\cR(\X)$] The diffeological group of $(\RR,+)$-principalbundles over $\X$.\item[$\Torus_{\alpha}$] The {\em torus of periods} of a closed form $\alpha$.\item[]\item[] {\bf \ChapterSD}\item[$\Psi$, $\Psi_\omega$] The {\em moment of paths}, the index $\omega$ is for {\em universal}.\item[$\psi$, $\psi_\omega$] The {\em $2$-points moment map}.\item[$\mu$, $\mu_\omega$] A {\em $1$-point moment map}.\item[$\theta$, $\theta_\omega$] The {\em lack of equivariance} of a moment map, a group cocycle in $\cG^*/\Gamma$.\item[$\sigma$, $\sigma_\omega$] The {\em Souriau class}, the class of the cocycles $\theta$, $\theta_\omega$.\end{itemize}
