\chapter*{Notation and Vocabulary}

%\renewcommand{\arraystretch}{0.965}.

\setlongtables
\begin{longtable}{rl}
%\begin{tabular}{rl}
\multicolumn{2}{c}{{\bf Diffeology and diffeological spaces}}\\


$\RR$& The {\em real numbers}.\\

$\NN$& The {\em natural integers}.\\

$\times$& The product of sets or the product of a number by\\
&\qquad something.\\

$\Def(f)$& The {\em set of definition\/} of the map $f$.\\

$\Val(f)$& The values of the map $f$.\\

$\Maps({\X},{\Y})$& The maps from ${\X}$ to ${\Y}$.\\ 

$\id_\X$& The identity map of $\X$.\\ 

$n$-domain& A (nonempty) open set of the vector space $\RR^n$.\\

domain& An arbitrary $n$-domain, for some $n$.\\

$\Domains(\RR^n)$& The  domains of $\RR^n$.\\ 

$\Domains$& The  domains of the $\RR^n$, $n$ running over $\NN$.\\ 

$\Powerset(\E)$& The {\em powerset} of the set $\E$, that is, the set
of all the\\
&\qquad subsets of the set $\E$.\\

$\dim(\U)$& The {\em dimension\/} of the domain $\U$, that is, if $\U
\subset \R^n$,\\
&\qquad $\dim(\U) = n$.\\

Parametrization& Any map defined from a domain to some set.\\

$\Param(\X)$& The parametrizations of a set ${\X}$.\\ 

$\Param(\U,\X)$& The parametrizations of a set $\X$ defined on a\\
&\qquad domain $\U$.\\ 

$\bmx$& A bold lower case letter, a constant map (here with\\
&\qquad value $x$).\\

$\D(\F)(s)$ or $\D(\F)_s$ or $\displaystyle{\partial r \over \partial s}$& For any smooth map $\F : s \mapsto r$ between real domains,\\
&\qquad the tangent linear map of ${\F}$ at the point $s$.\\

$\pr_k$& From a product, the projection on the $k$-th factor.\\

$\cD$& Refers to a diffeology.\\

$\D1, D2, D3$& Refers to the three axioms of diffeology.\\

$\cD(\U,\X)$& The plots, defined on $\U$, of the space $\X$.\\

$\cD(\X,\X')$ or $\Cinfty(\X,\X')$& The smooth maps from $\X$ to $\X'$.\\

$\Diff(\X,\X')$& The diffeomorphisms from $\X$ to $\X'$.\\

$\Set$, $\Diffeology$ etc.& Names of categories.\\

$\discrete{\cD}(\X)$& The discrete diffeology of ${\X}$.\\

$\X_\circ$& $\X$ equipped with the discrete diffeology.\\

$\coarse{\cD}(\X)$& The coarse diffeology of ${\X}$.\\

$\X_\bullet$& $\X$ equipped with the coarse diffeology.\\

$f^*(\cD')$& The pullback of the diffeology $\cD'$ by $f$.\\

$\coprod_{i \in \cI} \X_i$& The sum (coproduct) of diffeological spaces.\\

$\class(x)$& The equivalence class of $x$.\\

$f_*(\cD)$& The pushforward of the diffeology $\cD$ by $f$.\\

$\prod_{i \in \cI} \X_i$& The product of diffeological spaces.\\

$\Paths(\X)$& The space of paths in $\X$, that is, $\Cinfty(\R,\X)$.\\

$\DPaths{k}(\X)$& The space of iterated $k$-paths in $\X$, that is,\\
&\qquad $\Cinfty(\R,\DPaths{k-1}(\X))$.\\

$\delta x$& Any vector of $\RR^n$, when $x$ is a generic point in an\\
&\qquad $n$-domain.\\

$\gen{\cF}$& The diffeology generated by a family $\cF$ of\\
&\qquad parametrizations.\\

$\Nebula(\cF)$& The nebula of the family of parametrizations $\cF$.\\

$\dim(\X)$& The global dimension of $\X$, for a diffeological space.\\

$\Gen(\X)$& The generating families of the diffeology of $\X$.\\
\\
\multicolumn{2}{c}{{\bf Locality and diffeologies}}\\

$\Cinfty_\loc(\X,\X')$& The locally smooth maps from $\X$ to $\X'$.\\

$\Diff_\loc(\X,\X')$& The local diffeomorphisms from $\X$ to $\X'$.\\

$=_\loc$& Is equal/coincides locally.\\

$\germ$& The germ of a map.\\

D-open& Open for the D-topology.\\

$\dim_x(\X)$& The pointwise dimension of $\X$ at $x$.\\
\\
\multicolumn{2}{c}{{\bf Diffeological vector spaces}}\\

$\DLin(\E,\E')$& The smooth linear maps from $\E$ to $\E'$.\\

$\cH$, $\cH_\RR$, $\cH_\CC$& The standard Hilbert space, on $\RR$, on $\CC$.\\

$\X \cdot \Y$& The scalar or Hermitian product.\\
\\
\multicolumn{2}{c}{{\bf Modeling spaces, manifolds, etc.}}\\

$\cS_\RR$, $\cS_\CC$& The infinite Hilbert sphere, in $\cH_\RR$ or $\cH_\CC$.\\

$\cP$, $\cP_\CC$& The infinite projective space.\\

$\HH_n$, $\KK_n$& The half $n$-space or the $n$-corner.\\

\pagebreak
\multicolumn{2}{c}{{\bf Homotopy of diffeological spaces}}\\

$\0(\gamma)$, $\1(\gamma)$, $\ends(\gamma)$& Starting point, ending point, and ends of a path $\gamma$.\\

$\Paths(\X,\A,\B)$& The paths in $\X$, starting in $\A$ and ending in $\B$.\\

$\Loops(\X)$& Loops in $\X$.\\

$\Loops(\X,x)$, $\Loops_n(\X,x)$& Loops and iterated loops in $\X$, based at $x$.\\

$\gamma \vee \gamma'$& Concatenation of paths.\\

$\rev(\gamma)$& The reverse path of $\gamma$.\\

$\stPaths$, $\stLoops$& Stationary paths, loops.\\

$\gamma^\star$& Smashed path, making it stationary.\\

$\pi_0(\X)$, $\pi_k(\X,x)$& The components of $\X$ and $k$-th-homotopy groups\\
&\qquad based at $x$.\\

$\pi_k(\X,\A,x)$& Relative homotopy pointed-sets/groups.\\

$\Pi(\X)$& The Poincar\'e groupoid of $\X$.\\

\\

\multicolumn{2}{c}{{\bf Cartan-De Rham calculus}}\\

${\E}*{\F}$ or $\Lin(\E,\F)$& The space of linear maps from ${\E}$ to ${\F}$.\\

${\E}^*$& Dual vector space ${\E}$, that is, ${\E}*\RR$.\\

Vector --- Covector& An element of a vector space or its dual.\\
%Vector --- Covector& An element of a vector space --- An element of\\\n
%&\qquad its dual.\\\n

$\otimes$, $\wedge$& Tensor product, exterior product.\\

$\sgn(\sigma)$& Signature of the permutation $\sigma$.\\

$\DLin(\E,\F)$, ${\E}^*_\infty$& Smooth linear maps, smooth dual.\\

$\M^*(\A)$& Pullback of a covariant tensor $\A$ by a linear map $\M$.\\

$\vol$, $\vol_\cB$& Volume, canonical volume of a basis $\cB$.\\

$\Lambda^k({\E})$& The space of all $k$-linear forms on a vector space ${\E}$.\\

$\DForms^k(\X)$& The space of differentiable $k$-forms on a
diffeological\\
&\qquad space $\X$.\\

$\Forms^{k}_x(\X)$& The space of the values of the $k$-forms of $\X$
at the\\
&\qquad point $x$.\\

$\alpha(\P)$& The value of the differential form $\alpha$ on the plot $\P$.\\

$f^*(\alpha)$& The pullback of the differential form $\alpha$ by the\\
&\qquad smooth map $f$.\\

$d\alpha$& The {\em exterior derivative} of the differential form $\alpha$.\\

$\theta$& The canonical $1$-form on $\RR$ or on an irrational torus\\
&\qquad $\RR/\Gamma$.\\

$\delta \P$, $\delta\sigma$, $\delta c$& The {\em variation} of a plot $\P$, a cube $\sigma$, a chain $c$.\\

$\alpha \rfloor \delta \sigma$& The {\em contraction} of a $k$-form
$\alpha$ by a variation of a\\
&\qquad cube $\sigma$, a differential $(k-1)$-form on $\Dom(\sigma)$.\\

$i_{\F}$& The {\em contraction} by an arc of diffeomorphisms $\F$.\\

$\DLie_\F$& The Lie derivative by an arc of diffeomorphisms $\F$.\\

$\DCubes{p}(\X)$, $\DChains{p}(\X)$& The smooth $p$-cubes, $p$-chains, in $\X$.\\

$\ZG_\star(\X,\cdot)$, $\ZG^\star(\X,\cdot)$& The cycles/cocycles groups with coefficients.\\

$\BG_\star(\X,\cdot)$, $\BG^\star(\X,\cdot)$& The boundary/coboundary groups with coefficients.\\

$\HG_\star(\X,\cdot)$, $\HG^\star(\X,\cdot)$& The homology/cohomology groups with coefficients.\\

$\displaystyle{\int_\sigma\alpha}$, $\displaystyle\int_c\alpha$& The
%integral of a differential form on a cube/cubic-chain.\\\n
integral of a differential form on a\\
&\qquad cube/cubic-chain.\\

Magma& A set $\A$ equipped with an internal operation $\star$.\\

$\CHK$& The Chain-Homotopy operator.\\

$\eL(\gamma)$, $\eR(\gamma)$& Pre- or post-concatenation by $\gamma$.\\
\\
\multicolumn{2}{c}{{\bf Diffeological groups}}\\

$\DHom(\G,\G')$& The smooth homomorphisms from $\G$ to $\G'$.\\

$\Ab{\G}$& The Abelianized group $\G/[\G,\G]$.\\

$\eR(x)$ or $\hat x$& The orbit map from $\G$ to $\X$.\\

$\eL(g)$, $\eR(g)$& The left/right multiplication by $g$ in $\G$.\\

$\Ad(g)$& The adjoint action of $g$ on $\G$, $\Ad(g)\! =\! \eL(g)\! \circ\! \eR(g^{-1})$.\\

{\em Coset}& The left or right orbits of a subgroup in a group.\\

$\cG^*$& The {\em space of momenta} of a diffeological group $\G$.\\

$\Ad_*$, $\Ad_*^{\Gamma,\theta}$& The coadjoint action of $g$ on
$\cG^*$, on $\cG^*/\Gamma$ with\\
&\qquad cocycle $\theta$.\\
\\
\multicolumn{2}{c}{{\bf Diffeological fiber bundles}}\\

$\source$, $\target$, $\car$& The source, target and characteristic
maps on a\\
&\qquad groupoid.\\

$f^*(\T)$& The pullback by $f$ of the total space $\T$ of a projection.\\
%&\qquad projection.\\\n

$\T \times_\G \E$ or $\T \times_\rho \E$& An associate bundle.\\

$\Theta$& A connection on a principal bundle.\\

$\HorPaths(\Y,\Theta)$& The horizontal paths in $\Y$, for the connection $\Theta$.\\

$\T_{\Gamma}$& A {\em torus} $\RR^n/\Gamma$, $\Gamma$ discrete
(diffeologically) and\\
&\qquad generator of $\RR^n$.\\

%$\cR(\X)$& The diffeological group of $(\RR,+)$-principal\n
%bundles\\\n
%&\qquad over $\X$.\\\n
$\cR(\X)$& The group of $(\RR,+)$-principal
bundles over $\X$.\\

$\Torus_{\alpha}$& The {\em torus of periods} of a closed form $\alpha$.\\
\\
\multicolumn{2}{c}{{\bf Symplectic diffeology}}\\

$\Psi$, $\Psi_\omega$& The {\em moment of paths}, the index $\omega$ is for {\em universal}.\\

$\psi$, $\psi_\omega$& The {\em $2$-points moment map}.\\

$\mu$, $\mu_\omega$& A {\em $1$-point moment map}.\\

$\theta$, $\theta_\omega$& The {\em lack of equivariance} of a moment
map, a group\\
&\qquad cocycle in $\cG^*/\Gamma$.\\

$\sigma$, $\sigma_\omega$& The {\em Souriau class}, the class of the cocycles $\theta$, $\theta_\omega$.\\
%\end{tabular}
\end{longtable}
